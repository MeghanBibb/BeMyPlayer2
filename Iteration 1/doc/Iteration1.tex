\section{Project Vision}
Be My Player 2 is an interactive, gaming centric social media created to connect gamers with one another in multiple ways. Users will have the capability to find one another based on gaming preferences, search motivation, and personal information inputted through a questionnaire. Users will have several customization options regarding their profile including, gaming platform, favorite genre of play, and avatar selection. A number of quality features will be included such as a contact page and forgot password functionality. The system will offer the option to pay for preferential treatment in the find page of other users and encrypt user payment data. The system will also let users match with one another upon choosing and allow message functionality between the two. A comprehensive page of matches between other users will display other user data to the user.

\addtitle{Use Case 001}{Make Account} 
\addfield{Scope:}{User Database}
\addfield{Level:}{User goal}

	\item The System adds the User to the User Database.

}

		\begin{enumerate}
		\item[1.] User restarts System and returns to step 1.
		\end{enumerate}
	\item[*b] If at any time, User wishes to return to the Login page:
		\begin{enumerate}
		\item[1.] User selects option to return to the Login page.
		\item[2.] System returns to the Login page.
		\end{enumerate}
	\item[4.*] If the User submits an incomplete form:
		\begin{enumerate}
		\item[1.] System signals an error in the incomplete field(s) and rejects submission.
		\item[2.] User adjusts fields with errors.
		\item[3.] User completes all fields with errors.
		\item[4.] Return to step 4.
		\end{enumerate}
}

\newpage\subsection{Login to Account}\begin{usecase}
\addtitle{Use Case 004}{Login} 
\addfield{Scope:}{Be My Player 2 Application}
\addfield{Level:}{User Goal}
\addfield{Primary Actor:}{Registered User}
\additemizedfield{Stakeholders and Interests:}{
	\item User: Wants accurate and quick entry into the application after entering their correct account information
	along with the correct account password in, while also being secure.
	\item Company: Wants a simple and quick process to allow users to gain access to their account.
}
\additemizedfield{Preconditions:}{
	\item The User must have a Be My Player 2 account. 
	\item They must have a valid username and password. 
	\item The account is not a formerly deleted account.
	\item The User must be on the Log-In page.
} 
\additemizedfield{Postconditions:}{
	\item The User is logged into the system and is taken to the Home Page.
}
\addscenario{Main Success Scenario:}{
	\item User opens Be My Player 2 Application and is presented with the Log-In Page.
	\item User enters their username and password.
	\item User clicks the Log-In button.
	\item System finds the recognized account and sets their state to logged in.
	\item User is taken from Log-In Page to the Home Page.
}
\addscenario{Extensions:}{
	\item[3.a] User attempts to Log-In with one or more empty fields.
	\begin{enumerate}
		\item System does not check for an account with missing information.
		\item The User is prompted that information is missing and needs to be filled out completely.
	\end{enumerate}
	\item[4.a] System could not find an account with given username
	\begin{enumerate}
		\item The User is prompted that no account is registered under that username.
		\item The User is prompted to re-enter their information.
	\end{enumerate}
	\item[4.b] The User entered an invalid password for the given username.
	\begin{enumerate}
		\item The User is prompted that an incorrect password has been given
		\item The User is prompted to re-enter their information.
	\end{enumerate}
}

\addfield{Scope:}{Be My Player 2 Application}
\addfield{Level:}{User Goal}
\addfield{Primary Actor:}{Registered User}

	\item The User has their payment info stored on the System.

	\item The User is viewing the Edit Account Settings Page.
	\item The User clicks on the “Upgrade Account” button.
	\item The User is asked to enter his/her password to validate changes.
	\item The User inserts credit card number, expiration date, and cvv number.

	\begin{enumerate}

		\item The User returns to the Account Settings Page.
	\end{enumerate}

		\item The System prompts the user of an incorrect password.
		\item The User is taken back to the Account Settings page.

		\item The System prompts the User that one or more fields have been left blank, and to try again.

	\end{enumerate}
	\item[5.c] The entered expiration date month is not between 1 and 12.
	\begin{enumerate}
		\item The System prompts the User than an invalid month has been entered, and to try again.
	\end{enumerate}
	\item[5.d] The entered expiration date year is not a 2 digit number
	\begin{enumerate}
		\item The System prompts the User than an invalid year has been entered, and to try again.
	\end{enumerate}
	\item[5.e] The entered expiration date is past expiration.
	\begin{enumerate}
		\item The System prompts the User than an expired credit card has been entered.
		\item The User is taken back to the Account Settings page.
	\end{enumerate}
	\item[5.f] The entered cvv number is not a 3 digit number.
	\begin{enumerate}
		\item The System prompts the User than an invalid cvv number has been entered, and to try again.
	\end{enumerate}

\newpage\subsection{Block User}\begin{usecase}
	\addtitle{Use Case 011}{Block User} 
	\addfield{Scope:}{User Database}
	\addfield{Level:}{User goal}
	\addfield{Primary Actor:}{User}
	\additemizedfield{Stakeholders and Interests:}{
		\item User: Wants to block other user from viewing their profile, messaging them, and seeing in the matching pool.
		\item Company: Wants to ensure positive user experience by enabling users to prevent unwanted access by secondary user.
		\item Blocked User: Wants to continue using application without knowledge of users who have blocked them.
	\additemizedfield{Preconditions:}{
		\item Both users must have a Be My Player 2 account. 
		\item The Application User must be logged in to their account. 
		\item The Blocked User account must not be a deleted account.
	} 
	\additemizedfield{Postconditions:}{
		\item The Blocked User is blacklisted from the Application User’s match pool and added to their list of blocked users. 
		\item The Application User’s profile is blacklisted from the Blocked User’s match pool. 
		\item All communications between the Blocked User and Application User are removed. 
		\item The Application user can no longer view the Blocked User’s Profile and the Blocked User can no longer view the Application user’s profile.
	}
	\addscenario{Main Success Scenario:}{
		\item Application User is viewing another user’s profile.
		\item Application User clicks on “Block”.
		\item The System sends the Application user a confirmation form.
		\item Application User clicks “yes” to block user. 
		\item Application User is returned to the previous page.
		
	}
	\addscenario{Extensions:}{
		\item[*a] If at any time, the System fails:
		\begin{enumerate}
			\item[1.] The user is logged out and the block is not  updated with the System. 
		\end{enumerate}
		\item[4a.] User clicks “no” to not block user.
		\begin{enumerate}
			\item User is returned to the other user’s profile page.
		\end{enumerate}
	}
\end{usecase}
